<div id="ritual-container">
  <canvas id="scene"></canvas>
</div>
<script>
const canvas = document.getElementById("scene");
const ctx = canvas.getContext("2d");
let stars = [], loaded = 0;
const imgs = { left: new Image(), right: new Image(), chalice: new Image() };

// Файлы из папки Netlify
imgs.left.src = "2.png";
imgs.right.src = "3.png";
imgs.chalice.src = "6.png";

Object.values(imgs).forEach(img => {
  img.onload = () => { loaded++; };
});

function resize() {
  canvas.width = window.innerWidth;
  canvas.height = window.innerHeight;
  makeStars();
}

function makeStars() {
  stars = [];
  for (let i = 0; i < 70; i++) {
    stars.push({ x: Math.random()*canvas.width, y: Math.random()*canvas.height, r: Math.random()*1.2, a: Math.random(), s: Math.random()*0.02 });
  }
}

function draw(time) {
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  ctx.fillStyle = "#000";
  ctx.fillRect(0, 0, canvas.width, canvas.height);
  
  stars.forEach(s => {
    ctx.fillStyle = `rgba(255,255,255,${Math.abs(Math.sin(time*s.s))*s.a})`;
    ctx.beginPath(); ctx.arc(s.x, s.y, s.r, 0, Math.PI*2); ctx.fill();
  });

  if (loaded >= 3) {
    const scale = Math.min(canvas.width/1100, canvas.height/700);
    const midX = canvas.width / 2;
    const midY = canvas.height / 2;

    ctx.globalAlpha = 0.9;
    ctx.drawImage(imgs.left, midX - 480*scale, midY - 260*scale, 380*scale, 550*scale);
    ctx.drawImage(imgs.right, midX + 110*scale, midY - 240*scale, 320*scale, 500*scale);
    ctx.globalAlpha = 1;
    
    const cW = 200 * scale; const cH = 280 * scale;
    ctx.drawImage(imgs.chalice, midX - cW/2, midY - cH/2, cW, cH);
    
    const pulse = 0.8 + Math.sin(time*0.005)*0.1;
    const g = ctx.createRadialGradient(midX, midY - 40*scale, 0, midX, midY - 40*scale, 150*scale*pulse);
    g.addColorStop(0, `rgba(0,180,255,${0.35*pulse})`);
    g.addColorStop(1, "transparent");
    ctx.fillStyle = g; ctx.fillRect(0,0,canvas.width,canvas.height);
  }
  requestAnimationFrame(draw);
}
window.addEventListener("resize", resize);
resize(); requestAnimationFrame(draw);
</script>
