<script>
  // Устанавливаем дедлайн: 6 апреля 2026
  const deadlineDate = new Date(2026, 3, 6, 23, 59, 59).getTime();

  function updateLegalCountdown() {
    const now = new Date().getTime();
    const gap = deadlineDate - now;
    if (gap <= 0) {
      document.getElementById("countdown-legal-updated").innerHTML = "СРОК ИСТЕК / SCADUTO";
      return;
    }
    const d = Math.floor(gap / (1000 * 60 * 60 * 24));
    const h = Math.floor((gap % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
    const m = Math.floor((gap % (1000 * 60 * 60)) / (1000 * 60));
    const s = Math.floor((gap % (1000 * 60)) / 1000);

    const f = (n) => String(n).padStart(2, '0');
    document.getElementById("countdown-legal-updated").innerHTML = f(d) + ":" + f(h) + ":" + f(m) + ":" + f(s);
  }

  setInterval(updateLegalCountdown, 1000);
  updateLegalCountdown();
</script>
