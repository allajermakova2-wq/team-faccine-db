  <style>
    html, body {

  <canvas id="kidsTrainCanvas"></canvas>
</div>

<script>
const canvas = document.getElementById("kidsTrainCanvas");
const ctx = canvas.getContext("2d");
let dpr = window.devicePixelRatio || 1;
let cssW = 0;
let cssH = 185;

const wagons = [
  { code: "AF",    color: ["#ffb09d", "#da6155"] },
  { code: "PS",    color: ["#a7efb7", "#56b56e"] },
  { code: "CE/95", color: ["#d0c0ff", "#9170e5"] },
  { code: "IT",    color: ["#ff9b8e", "#d8574b"] },
  { code: "RU",    color: ["#9dc9ff", "#5e90e0"] },
  { code: "UA",    color: ["#fff0b0", "#8dc7ff"] },
  { code: "EN",    color: ["#a8e3ff", "#63b2dd"] },
  { code: "FR",    color: ["#aabfff", "#687fd8"] },
  { code: "DE",    color: ["#ffe59d", "#ddb14c"] },
  { code: "LV",    color: ["#e4b9ff", "#aa71dd"] },
  { code: "HE",    color: ["#aef4d7", "#59b995"] }
];

let trainX = -1500;
let wheelAngle = 0;
let smoke = [];

function resize() {
  dpr = window.devicePixelRatio || 1;
  cssW = canvas.clientWidth || window.innerWidth;
  cssH = 185;

  canvas.width = cssW * dpr;
  canvas.height = cssH * dpr;

  ctx.setTransform(1, 0, 0, 1, 0, 0);
  ctx.scale(dpr, dpr);
}
window.addEventListener("resize", resize);
resize();

function drawRoundedRect(x, y, w, h, r) {
  ctx.beginPath();
  ctx.moveTo(x + r, y);
  ctx.arcTo(x + w, y, x + w, y + h, r);
  ctx.arcTo(x + w, y + h, x, y + h, r);
  ctx.arcTo(x, y + h, x, y, r);
  ctx.arcTo(x, y, x + w, y, r);
  ctx.closePath();
}

function drawWheel(x, y, r) {
  ctx.save();
  ctx.translate(x, y);
  ctx.rotate(wheelAngle);

  const g = ctx.createRadialGradient(-2, -2, 1, 0, 0, r);
  g.addColorStop(0, "#ffffff");
  g.addColorStop(0.16, "#a0a0a0");
  g.addColorStop(0.45, "#3d3d3d");
  g.addColorStop(1, "#111");

  ctx.fillStyle = g;
  ctx.beginPath();
  ctx.arc(0, 0, r, 0, Math.PI * 2);
  ctx.fill();

  ctx.strokeStyle = "rgba(255,255,255,0.28)";
  ctx.lineWidth = 1.1;
  ctx.beginPath();
  ctx.arc(0, 0, r - 3, 0, Math.PI * 2);
  ctx.stroke();

  ctx.strokeStyle = "rgba(255,255,255,0.20)";
  for (let i = 0; i < 6; i++) {
    ctx.rotate(Math.PI / 3);
    ctx.beginPath();
    ctx.moveTo(0, 0);
    ctx.lineTo(r - 3, 0);
    ctx.stroke();
  }

  ctx.fillStyle = "#8d8d8d";
  ctx.beginPath();
  ctx.arc(0, 0, 2.7, 0, Math.PI * 2);
  ctx.fill();

  ctx.restore();
}

function drawRail(canvasW) {
  ctx.strokeStyle = "#927247";
  ctx.lineWidth = 4;
  ctx.beginPath();
  ctx.moveTo(0, 152);
  ctx.lineTo(canvasW, 152);
  ctx.stroke();

  ctx.strokeStyle = "#5d472d";
  ctx.lineWidth = 2;
  ctx.beginPath();
  ctx.moveTo(0, 158);
  ctx.lineTo(canvasW, 158);
  ctx.stroke();

  for (let i = 0; i < canvasW; i += 33) {
    ctx.strokeStyle = "rgba(255,255,255,0.18)";
    ctx.lineWidth = 2;
    ctx.beginPath();
    ctx.moveTo(i, 142);
    ctx.lineTo(i, 163);
    ctx.stroke();
  }
}

function drawTitle() {
  ctx.fillStyle = "#dfff6e";
  ctx.font = "bold 18px Trebuchet MS";
  ctx.textAlign = "left";
  ctx.fillText("🚂 Le indagini delle Faccine — linguaggi del viaggio", 18, 24);
}

function drawSmoke() {
  smoke.forEach(p => {
    ctx.globalAlpha = p.a;
    ctx.fillStyle = "#ffffff";
    ctx.beginPath();
    ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
    ctx.fill();
  });
  ctx.globalAlpha = 1;
}

function updateSmoke(pipeX, pipeY) {
  if (Math.random() > 0.35) {
    smoke.push({
      x: pipeX,
      y: pipeY,
      r: 4 + Math.random() * 3,
      a: 0.55 + Math.random() * 0.2,
      vx: -1.2 - Math.random() * 0.7,
      vy: -0.4 - Math.random() * 0.8
    });
  }

  smoke.forEach(p => {
    p.x += p.vx;
    p.y += p.vy;
    p.r += 0.18;
    p.a -= 0.012;
  });

  smoke = smoke.filter(p => p.a > 0);
}

function drawEngine(x, y) {
  const w = 195, h = 72;

  const g = ctx.createLinearGradient(x, y, x, y + h);
  g.addColorStop(0, "#79ace8");
  g.addColorStop(0.55, "#4774b2");
  g.addColorStop(1, "#27446c");
  ctx.fillStyle = g;
  drawRoundedRect(x, y, w, h, 18);
  ctx.fill();

  ctx.fillStyle = "rgba(255,255,255,0.13)";
  drawRoundedRect(x + 8, y + 5, w - 16, 18, 10);
  ctx.fill();

  const cabG = ctx.createLinearGradient(x, y - 18, x, y + 20);
  cabG.addColorStop(0, "#688fca");
  cabG.addColorStop(1, "#3b5e94");
  ctx.fillStyle = cabG;
  drawRoundedRect(x + 8, y - 18, 58, 38, 10);
  ctx.fill();

  ctx.fillStyle = "#e6ff79";
  drawRoundedRect(x + 18, y - 8, 26, 16, 4);
  ctx.fill();

  ctx.fillStyle = "#1b2c48";
  ctx.fillRect(x + 138, y - 26, 18, 28);
  ctx.fillRect(x + 133, y - 31, 28, 8);

  ctx.fillStyle = "#1e3456";
  ctx.beginPath();
  ctx.arc(x + w, y + h / 2, h / 2, -Math.PI / 2, Math.PI / 2);
  ctx.fill();

  ctx.fillStyle = "#ffe88a";
  ctx.beginPath();
  ctx.arc(x + w + 16, y + h / 2, 7, 0, Math.PI * 2);
  ctx.fill();

  ctx.shadowColor = "rgba(255,232,138,0.8)";
  ctx.shadowBlur = 16;
  ctx.beginPath();
  ctx.arc(x + w + 16, y + h / 2, 7, 0, Math.PI * 2);
  ctx.fill();
  ctx.shadowBlur = 0;

  ctx.strokeStyle = "#8a6a42";
  ctx.lineWidth = 4;
  ctx.beginPath();
  ctx.moveTo(x + w + 22, y + h / 2);
  ctx.lineTo(x + w + 32, y + h / 2);
  ctx.stroke();

  ctx.fillStyle = "#fff8e0";
  ctx.font = "bold 14px Georgia";
  ctx.textAlign = "center";
  ctx.fillText("Le indagini delle Faccine", x + 98, y + 43);

  drawWheel(x + 40,  y + h + 5, 11);
  drawWheel(x + 138, y + h + 5, 11);

  return {
    pipeX: x + 147,
    pipeY: y - 20
  };
}

function drawWagon(x, y, w, code, c1, c2) {
  const h = 60;

  ctx.fillStyle = "#555";
  drawRoundedRect(x - 4, y - 4, w + 8, 10, 5);
  ctx.fill();

  const g = ctx.createLinearGradient(x, y, x, y + h);
  g.addColorStop(0, c1);
  g.addColorStop(1, c2);
  ctx.fillStyle = g;
  drawRoundedRect(x, y, w, h, 13);
  ctx.fill();

  ctx.fillStyle = "rgba(255,255,255,0.14)";
  drawRoundedRect(x + 7, y + 5, w - 14, 13, 7);
  ctx.fill();

  ctx.fillStyle = "rgba(0,0,0,0.10)";
  drawRoundedRect(x + 8, y + 34, w - 16, 14, 7);
  ctx.fill();

  ctx.fillStyle = "rgba(255,255,255,0.35)";
  ctx.beginPath(); ctx.arc(x + 16, y + 18, 7, 0, Math.PI * 2); ctx.fill();
  ctx.beginPath(); ctx.arc(x + w - 16, y + 18, 7, 0, Math.PI * 2); ctx.fill();

  ctx.fillStyle = "#fff";
  ctx.font = "bold 16px Arial";
  ctx.textAlign = "center";
  ctx.fillText(code, x + w / 2, y + 40);

  drawWheel(x + 18,     y + h + 5, 9);
  drawWheel(x + w - 18, y + h + 5, 9);
}

function animate() {
  ctx.clearRect(0, 0, cssW, cssH);

  drawTitle();
  drawRail(cssW);

  const engine = drawEngine(trainX, 68);

  updateSmoke(engine.pipeX, engine.pipeY);
  drawSmoke();

  let currentX = trainX - 18;

  wagons.forEach(wagon => {
    const vW = wagon.code === "CE/95" ? 108 : 86;
    currentX -= (vW + 14);

    ctx.strokeStyle = "#8a6a42";
    ctx.lineWidth = 4;
    ctx.beginPath();
    ctx.moveTo(currentX + vW, 110);
    ctx.lineTo(currentX + vW + 14, 110);
    ctx.stroke();

    drawWagon(currentX, 83, vW, wagon.code, wagon.color[0], wagon.color[1]);
  });

  trainX += 1.8;
  wheelAngle += 0.15;

  if (trainX > cssW + 1700) {
    trainX = -1300;
    smoke = [];
  }

  requestAnimationFrame(animate);
}

animate();
</script>
</body>
</html>
