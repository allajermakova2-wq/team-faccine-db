<script>
const canvas = document.getElementById('c');
const ctx = canvas.getContext('2d');

function resize(){
  canvas.width = innerWidth;
  canvas.height = innerHeight;
}
resize();
onresize = resize;

/* страны */
const countries = [
"UNITED STATES","CANADA","GREAT BRITAIN","GERMANY","FRANCE",
"NETHERLANDS","BELGIUM","ITALY","SPAIN","SWITZERLAND","POLAND",
"BRAZIL","SOUTH AFRICA","INDIA","SINGAPORE","JAPAN",
"SOUTH KOREA","UAE","AUSTRALIA","ARGENTINA"
];

/* бегущая строка */
const track = document.getElementById("track");

function addCountries(){
  countries.forEach(c=>{
    const el = document.createElement("span");
    el.className="country";
    el.textContent = c + " — SYSTEM ACTIVE";
    track.appendChild(el);
  });
}
addCountries();
addCountries();

/* матрица */
const chars="ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";
let size=16;
let cols=Math.floor(canvas.width/size);
let drops=new Array(cols).fill(1);

function drawMatrix(){
  ctx.fillStyle="rgba(0,0,0,0.07)";
  ctx.fillRect(0,0,canvas.width,canvas.height);

  ctx.fillStyle="#00ff66";
  ctx.font=size+"px monospace";

  for(let i=0;i<drops.length;i++){
    const text=chars[Math.floor(Math.random()*chars.length)];
    ctx.fillText(text,i*size,drops[i]*size);

    if(drops[i]*size>canvas.height && Math.random()>0.97) drops[i]=0;
    drops[i]++;
  }
}

/* карта */
const nodes=[];

function buildMap(){
  nodes.length=0;
  for(let i=0;i<40;i++){
    nodes.push({
      x:Math.random()*canvas.width,
      y:Math.random()*canvas.height,
      p:Math.random()*Math.PI*2
    });
  }
}
buildMap();

function drawMap(){
  nodes.forEach((n,i)=>{
    let glow=0.4+Math.sin(Date.now()*0.002+n.p)*0.3;

    ctx.fillStyle="rgba(120,220,255,"+glow+")";
    ctx.beginPath();
    ctx.arc(n.x,n.y,2,0,Math.PI*2);
    ctx.fill();

    if(i>0){
      const p=nodes[i-1];
      ctx.strokeStyle="rgba(120,220,255,0.1)";
      ctx.beginPath();
      ctx.moveTo(n.x,n.y);
      ctx.lineTo(p.x,p.y);
      ctx.stroke();
    }
  });
}

/* импульсы */
let pulses=[];

function spawnPulse(){
  if(Math.random()<0.05){
    let a=Math.floor(Math.random()*nodes.length);
    let b=Math.floor(Math.random()*nodes.length);
    pulses.push({a,b,t:0});
  }
}

function drawPulses(){
  for(let i=pulses.length-1;i>=0;i--){
    let p=pulses[i];
    p.t+=0.02;

    let n1=nodes[p.a];
    let n2=nodes[p.b];

    let x=n1.x+(n2.x-n1.x)*p.t;
    let y=n1.y+(n2.y-n1.y)*p.t;

    ctx.fillStyle="#fff";
    ctx.beginPath();
    ctx.arc(x,y,2,0,Math.PI*2);
    ctx.fill();

    if(p.t>=1) pulses.splice(i,1);
  }
}

/* эффекты */
const flash=document.getElementById("flash");
const wave=document.getElementById("wave");

function alertFlash(){
  flash.style.opacity=1;
  setTimeout(()=>flash.style.opacity=0,200);
}

function breach(){
  wave.style.transform="translateX(100%)";
  setTimeout(()=>wave.style.transform="translateX(-100%)",800);
}

/* глитчи */
function glitch(){
  const els=document.querySelectorAll(".country");
  const el=els[Math.floor(Math.random()*els.length)];

  el.style.color="#ff3333";

  if(Math.random()<0.5) breach();
  else alertFlash();

  setTimeout(()=>el.style.color="#00ff66",300);
}

/* цикл */
function loop(){
  drawMatrix();
  drawMap();
  spawnPulse();
  drawPulses();
  requestAnimationFrame(loop);
}

loop();

setInterval(glitch,2500);
</script>
