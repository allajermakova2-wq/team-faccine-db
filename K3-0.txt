  <script>
    const stars = document.getElementById("stars");
    const matrixRain = document.getElementById("matrixRain");
    const chars = "01アイウエオカキクケコサシスセソABCDEFGHIJKLMNOPQRSTUVWXYZ<>[]{}#$%&*+-";

    function createStars() {
      const count = Math.floor(window.innerWidth / 10);

      for (let i = 0; i < count; i++) {
        const s = document.createElement("div");
        s.className = "star";

        const size = Math.random() * 2.8 + 0.7;
        s.style.width = size + "px";
        s.style.height = size + "px";
        s.style.left = Math.random() * 100 + "%";
        s.style.top = Math.random() * 100 + "%";
        s.style.animationDuration = (1.6 + Math.random() * 4.2) + "s";
        s.style.animationDelay = (Math.random() * 4) + "s";
        stars.appendChild(s);
      }
    }

    function buildColumnText(lines = 26) {
      let out = "";
      for (let i = 0; i < lines; i++) {
        out += chars[Math.floor(Math.random() * chars.length)] + "\n";
      }
      return out;
    }

    function createRain() {
      const colCount = Math.floor(window.innerWidth / 28);

      for (let i = 0; i < colCount; i++) {
        const col = document.createElement("div");
        col.className = "column";
        col.textContent = buildColumnText(18 + Math.floor(Math.random() * 18));
        col.style.left = (i * 28 + Math.random() * 10) + "px";
        col.style.animationDuration = (8 + Math.random() * 12) + "s";
        col.style.animationDelay = (-Math.random() * 18) + "s";
        col.style.fontSize = (13 + Math.random() * 6) + "px";
        col.style.lineHeight = col.style.fontSize;
        matrixRain.appendChild(col);
      }
    }

    createStars();
    createRain();

    function randomGlitchPulse(el) {
      const min = 400;
      const max = 1800;

      function trigger() {
        el.classList.remove("glitching");
        void el.offsetWidth;
        el.classList.add("glitching");

        const duration = 120 + Math.random() * 380;

        setTimeout(() => {
          el.classList.remove("glitching");
          setTimeout(trigger, min + Math.random() * (max - min));
        }, duration);
      }

      trigger();
    }

    randomGlitchPulse(document.getElementById("leftDigit"));
    randomGlitchPulse(document.getElementById("rightDigit"));

    const panel = document.querySelector(".panel");
    setInterval(() => {
      panel.style.opacity = "0.94";
      panel.style.transform = "translateY(0.5px)";
      setTimeout(() => {
        panel.style.opacity = "1";
        panel.style.transform = "translateY(0)";
      }, 70);
    }, 2200 + Math.random() * 1400);
