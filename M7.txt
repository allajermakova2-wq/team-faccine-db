const canvas = document.getElementById('matrixCanvas');

const fontSize = 16;

let drops = [];

let pulses = [];
let learningLevel = 0;
let repeatedWord = null;
let repeatedWordTimer = 0;
let nodes = [];
let chainLinks = [];

function resizeCanvas() {
  canvas.width = innerWidth;
  canvas.height = innerHeight;
  columns = Math.floor(canvas.width / fontSize);
  drops = Array.from({ length: columns }, () => Math.floor(Math.random() * (canvas.height / fontSize)));
  buildWorld();
}
resizeCanvas();
addEventListener('resize', resizeCanvas);

function dnsNode(x, y, label, weight = 1) {
  return {
    x: canvas.width * x,
    y: canvas.height * y,
    label,
    weight,
    phase: Math.random() * Math.PI * 2
  };
}

function buildWorld() {
  nodes = [

function randomChar() {
  return chars[Math.floor(Math.random() * chars.length)];
}

  "UNITED STATES",
  "CANADA",

function pickOverlayWord() {
  const priorityWords = [
    "UNITED STATES",
    "GERMANY",
    "FRANCE",
    "NETHERLANDS",
    "POLAND",
    "GREAT BRITAIN",
    "INDIA",
    "SINGAPORE",
    "JAPAN",
    "SOUTH KOREA",
    "TURKEY",
    "UAE",
    "AUSTRALIA"
  ];

  const boostedPool = [...countryWords, ...priorityWords, ...priorityWords, ...glitchWords];

  if (repeatedWord && Math.random() < learningLevel * 0.75) {
    return repeatedWord;
  }

  const chosen = boostedPool[Math.floor(Math.random() * boostedPool.length)];

  if (learningLevel > 0.45 && Math.random() < 0.35) {
    repeatedWord = chosen;
    repeatedWordTimer = 6 + Math.floor(Math.random() * 8);
  }

  return chosen;
}

function spawnOverlayWord() {
  const word = pickOverlayWord();
  overlays.push({
    text: word,
    x: Math.random() * Math.max(80, canvas.width - word.length * 14),
    y: Math.random() * Math.max(120, canvas.height - 60),
    life: 0,
    maxLife: 80 + Math.floor(Math.random() * 45)
  });
}

function spawnPulse() {
  if (Math.random() < 0.08 && chainLinks.length) {
    const [a, b] = chainLinks[Math.floor(Math.random() * chainLinks.length)];
    pulses.push({
      a, b, t: 0,
      speed: 0.018 + Math.random() * 0.022
    });
  }
}

function drawWorldGlowBands() {
  ctx.save();

  ctx.fillStyle = "rgba(80,140,255,0.06)";
  ctx.beginPath();
  ctx.ellipse(canvas.width * 0.56, canvas.height * 0.43, canvas.width * 0.12, canvas.height * 0.07, -0.08, 0, Math.PI * 2);
  ctx.fill();

  ctx.fillStyle = "rgba(120,210,255,0.05)";
  ctx.beginPath();
  ctx.ellipse(canvas.width * 0.82, canvas.height * 0.54, canvas.width * 0.12, canvas.height * 0.07, 0.12, 0, Math.PI * 2);
  ctx.fill();

  ctx.restore();
}

function drawEasternArc() {
  const t = Date.now() * 0.002;
  const alpha = 0.08 + 0.05 * (1 + Math.sin(t));

  ctx.save();
  ctx.lineWidth = 2;
  ctx.strokeStyle = `rgba(120,220,255,${alpha})`;
  ctx.shadowColor = 'rgba(120,220,255,0.45)';
  ctx.shadowBlur = 14;

  ctx.beginPath();
  ctx.moveTo(canvas.width * 0.56, canvas.height * 0.42);
  ctx.quadraticCurveTo(canvas.width * 0.71, canvas.height * 0.28, canvas.width * 0.88, canvas.height * 0.47);
  ctx.stroke();

  ctx.lineWidth = 1;
  ctx.strokeStyle = `rgba(255,255,255,${alpha * 0.75})`;
  ctx.beginPath();
  ctx.moveTo(canvas.width * 0.58, canvas.height * 0.44);
  ctx.quadraticCurveTo(canvas.width * 0.72, canvas.height * 0.33, canvas.width * 0.86, canvas.height * 0.49);
  ctx.stroke();
  ctx.restore();
}

function drawNetwork() {
  const t = Date.now() * 0.0018;
  ctx.save();
  ctx.lineCap = 'round';

  chainLinks.forEach(([ai, bi], idx) => {
    const a = nodes[ai], b = nodes[bi];
    if (!a || !b) return;

    const pulse = 0.14 + 0.14 * (1 + Math.sin(t * 2 + idx));
    const glow = 0.04 + 0.03 * (1 + Math.sin(t * 3 + idx * 0.7));

    ctx.strokeStyle = `rgba(120,220,255,${pulse})`;
    ctx.shadowColor = `rgba(120,220,255,${glow})`;
    ctx.shadowBlur = 10;
    ctx.lineWidth = 1.1;

    ctx.beginPath();
    ctx.moveTo(a.x, a.y);
    ctx.lineTo(b.x, b.y);
    ctx.stroke();
  });

  for (let i = pulses.length - 1; i >= 0; i--) {
    const p = pulses[i];
    const a = nodes[p.a], b = nodes[p.b];
    if (!a || !b) {
      pulses.splice(i, 1);
      continue;
    }

    p.t += p.speed;
    const x = a.x + (b.x - a.x) * p.t;
    const y = a.y + (b.y - a.y) * p.t;

    ctx.shadowBlur = 16;
    ctx.shadowColor = 'rgba(255,255,255,0.9)';
    ctx.fillStyle = 'rgba(255,255,255,0.95)';
    ctx.beginPath();
    ctx.arc(x, y, 2.2, 0, Math.PI * 2);
    ctx.fill();

    ctx.shadowBlur = 28;
    ctx.fillStyle = 'rgba(120,220,255,0.42)';
    ctx.beginPath();
    ctx.arc(x, y, 5, 0, Math.PI * 2);
    ctx.fill();

    if (p.t >= 1) pulses.splice(i, 1);
  }

  ctx.restore();
}

function drawNodes() {
  nodes.forEach(n => {
    const alpha = 0.18 + Math.abs(Math.sin(Date.now() * 0.002 + n.phase)) * 0.22 * n.weight;

    ctx.fillStyle = `rgba(120,220,255,${alpha})`;
    ctx.beginPath();
    ctx.arc(n.x, n.y, 2.2 * n.weight, 0, Math.PI * 2);
    ctx.fill();

    ctx.fillStyle = `rgba(255,255,255,${alpha * 0.30})`;
    ctx.beginPath();
    ctx.arc(n.x, n.y, 4.8 * n.weight, 0, Math.PI * 2);
    ctx.fill();
  });
}

function drawMatrix() {
  ctx.fillStyle = 'rgba(0,0,0,0.08)';
  ctx.fillRect(0, 0, canvas.width, canvas.height);

  drawWorldGlowBands();
  drawEasternArc();
  drawNetwork();
  drawNodes();

  ctx.font = fontSize + 'px monospace';

  for (let i = 0; i < drops.length; i++) {
    const isHack = Math.random() < (0.02 + learningLevel * 0.05);
    const text = randomChar();

    if (isHack) {
      ctx.fillStyle = Math.random() < 0.7 ? '#ff2a2a' : '#ffffff';
      ctx.shadowColor = '#ff2a2a';
    } else {
      ctx.fillStyle = '#00ff66';
      ctx.shadowColor = '#00ff66';
    }

    ctx.shadowBlur = 8;
    ctx.fillText(text, i * fontSize, drops[i] * fontSize);

    if (Math.random() < 0.003 + learningLevel * 0.008) {
      ctx.fillStyle = Math.random() < 0.5 ? '#ffffff' : '#ff2a2a';
      ctx.fillText('!', i * fontSize, drops[i] * fontSize);
    }

    if (drops[i] * fontSize > canvas.height && Math.random() > 0.975) drops[i] = 0;
    drops[i]++;
  }

  ctx.shadowBlur = 0;

  for (let i = overlays.length - 1; i >= 0; i--) {
    const o = overlays[i];
    o.life++;
    const alpha = Math.max(0, 1 - o.life / o.maxLife);
    const isGlitchWord = glitchWords.includes(o.text);

    ctx.fillStyle = isGlitchWord
      ? `rgba(255,60,60,${alpha * 0.26})`
      : `rgba(255,40,40,${alpha * 0.12})`;
    ctx.font = 'bold 26px monospace';
    ctx.fillText(o.text, o.x + 2, o.y);

    ctx.fillStyle = isGlitchWord
      ? `rgba(255,80,80,${alpha * 0.95})`
      : `rgba(0,255,102,${alpha * 0.92})`;
    ctx.shadowColor = isGlitchWord ? '#ff4d4d' : '#00ff66';
    ctx.shadowBlur = 12;
    ctx.fillText(o.text, o.x, o.y);
    ctx.shadowBlur = 0;

    if (o.life >= o.maxLife) overlays.splice(i, 1);
  }
}

function scrambleToText(el, finalText, duration = 1400) {
  let frame = 0;
  const totalFrames = Math.max(20, Math.floor(duration / 40));

  const interval = setInterval(() => {
    let output = '';
    for (let i = 0; i < finalText.length; i++) {
      if (finalText[i] === ' ') {
        output += ' ';
        continue;
      }

      const progress = frame / totalFrames;
      const revealIndex = Math.floor(progress * finalText.length);

      output += i < revealIndex
        ? finalText[i]
        : scramblePool[Math.floor(Math.random() * scramblePool.length)];
    }

    el.textContent = output;
    frame++;

    if (frame > totalFrames) {
      clearInterval(interval);
      el.textContent = finalText;
    }
  }, 40);
}

function initDecrypt() {
  countryEls.forEach((el, idx) => {
    setTimeout(() => scrambleToText(el, el.dataset.final, 1200 + Math.random() * 700), idx * 100);
  });
}

function triggerGlobalBreach() {
  countryEls.forEach(el => el.classList.add('global-breach'));
  setTimeout(() => countryEls.forEach(el => el.classList.remove('global-breach')), 1000);
}

function triggerChainReaction(originIndex) {
  const neighbors = [
    countryEls[originIndex - 2],
    countryEls[originIndex - 1],
    countryEls[originIndex + 1],
    countryEls[originIndex + 2]
  ].filter(Boolean);

  neighbors.forEach((node, idx) => {
    setTimeout(() => {
      node.classList.add('chain-react');
      setTimeout(() => node.classList.remove('chain-react'), 420);
    }, idx * 80);
  });
}

  alertWave.classList.remove('active');
  void alertWave.offsetWidth;
  alertWave.classList.add('active');
  triggerGlobalBreach();
  setTimeout(() => marqueeContainer.classList.remove('breach-mode'), 950);
}

function triggerAlertFlash() {
  alertFlash.classList.remove('active');
  void alertFlash.offsetWidth;
  alertFlash.classList.add('active');
}

function intensifyLearning(triggerWord) {
  learningLevel = Math.min(1, learningLevel + 0.12);

  if (["BREACH", "ALERT", "ACCESS"].includes(triggerWord)) {
    repeatedWord = triggerWord;
    repeatedWordTimer = 8 + Math.floor(Math.random() * 6);
    learningLevel = Math.min(1, learningLevel + 0.12);
  }
}

initDecrypt();

function loop() {
  drawMatrix();
  spawnPulse();
  requestAnimationFrame(loop);
}
loop();

setInterval(() => {
  const spawnChance = 0.88 + learningLevel * 0.08;
  if (Math.random() < spawnChance) spawnOverlayWord();

  if (repeatedWordTimer > 0) repeatedWordTimer--;
  else repeatedWord = null;

  learningLevel *= 0.992;
}, 1800);

setInterval(() => {
  const originIndex = Math.floor(Math.random() * countryEls.length);
  const el = countryEls[originIndex];
  if (!el) return;

  el.classList.add('glitch');
  const original = el.dataset.final;
  let chosenTrigger = null;
  let count = 0;

  const g = setInterval(() => {
    const triggerWord = glitchWords[Math.floor(Math.random() * glitchWords.length)];
    if (!chosenTrigger && Math.random() < 0.55) chosenTrigger = triggerWord;

    const hacked = original.split('').map(ch => {
      if (ch === ' ' || Math.random() > (0.18 + learningLevel * 0.12)) return ch;
      if (Math.random() < (0.25 + learningLevel * 0.18)) {
        const word = chosenTrigger || triggerWord;
        return word[Math.floor(Math.random() * word.length)];
      }
      return scramblePool[Math.floor(Math.random() * scramblePool.length)];
    }).join('');

    el.textContent = hacked;
    count++;

    if (count === 1 && chosenTrigger) triggerChainReaction(originIndex);

    if (chosenTrigger === "BREACH" && count === 1) {
      triggerBreachWave();
      intensifyLearning("BREACH");
    }

    if (chosenTrigger === "ALERT" && count === 1) {
      triggerAlertFlash();
      intensifyLearning("ALERT");
    }

    if (chosenTrigger === "ACCESS" && count === 1) {
      intensifyLearning("ACCESS");
    }

    if (count > 3) {
      clearInterval(g);
      el.textContent = original;
      el.classList.remove('glitch');
    }
  }, 70);
}, 2500);
